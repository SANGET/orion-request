import { useState } from 'react';
import { Playground } from 'docz'
import { urlParamsToQuery, getUrlParams } from '@mini-code/request/url-resolve'

# URL解析器

## Basic Usage

<Playground>
  {
    () => {
      const baseUrl = "http://127.0.0.1:55555";
      const params = {
        ID: '123',
        other: '321',
        url: 'https://google.com'
      }
      const defaultUrl = urlParamsToQuery({
        url: baseUrl,
        params,
        toBase64: false
      })
      const [urlRes, setUrl] = useState(defaultUrl);
      return (
        <div>
          <textarea
            className="form-control"
            defaultValue={defaultUrl}
            onChange={e => setUrl(e.target.value)} />
          <hr />
          <div>解析结果: {
            JSON.stringify(getUrlParams(null, urlRes, false))
          }</div>
        </div>
      )
    }
  }
</Playground>
