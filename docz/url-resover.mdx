import { useState } from 'react';
import { Playground } from 'docz'
import { getUrlParams } from '../src/url-resolve'

# URL解析器

<Playground>
  {
    () => {
      const defaultUrl = "http://127.0.0.1:55555/#/?_R=Chat&ID=qqq";
      const [urlRes, setUrl] = useState(defaultUrl);
      return (
        <div>
          <textarea
            className="form-control"
            defaultValue={defaultUrl}
            onChange={e => setUrl(e.target.value)} />
          <hr />
          <div>解析结果: {
            JSON.stringify(getUrlParams(null, urlRes))
          }</div>
        </div>
      )
    }
  }
</Playground>
